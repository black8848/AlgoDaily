class MyLinkedList {
public:
    struct ListNode {
        int val;
        ListNode* next;
        ListNode() : val(0), next(nullptr) {}
    };

    MyLinkedList() { Head = new ListNode(); }

    int get(int index) {
        if (index < 0) {
            return -1;
        }
        ListNode* p = Head->next;
        while (index-- && p != nullptr) {
            p = p->next;
        }
        return p ? p->val : -1;
    }

    void addAtHead(int val) {
        ListNode* p = new ListNode();
        p->next = Head->next;
        p->val = val;
        Head->next = p;
    }

    void addAtTail(int val) {
        ListNode* p = Head;
        ListNode* node = new ListNode();
        while (p->next != nullptr) {
            p = p->next;
        }
        node->val = val;
        node->next = nullptr;
        p->next = node;
    }

    void addAtIndex(int index, int val) {
        if (index < 0) {
            return;
        }
        ListNode* p = new ListNode();
        ListNode* node = new ListNode();
        p = Head;
        
        while (index--) {
            p = p->next;
            if (p == nullptr) {
                return;
            }
        }
        node->val = val;
        node->next = p->next;
        p->next = node;
    }

    void deleteAtIndex(int index) {
        if (index < 0) {
            return;
        }
        ListNode* p = Head;
        ListNode* node = new ListNode();
        while (index--) {
            p = p->next;
            if (p == nullptr) {
                return;
            }
        }
        node = p->next;
        if(p->next== nullptr){
            return;
        }
        p->next = p->next->next;
        delete node;
    }

private:
    ListNode* Head;
};
